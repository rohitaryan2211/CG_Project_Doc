167 // Get properties from the bufferView

178
179 // Go over all the bytes in the data at the correct place using the properties from above
180 unsigned int beginningOfData = byteOffset + accByteOffset;
181 unsigned int lengthOfData = count * 4 * numPerVert;
182 for (unsigned int i = beginningOfData; i < beginningOfData + lengthOfData; i)
183 {
184 unsigned char bytes[] = { data[i++], data[i++], data[i++], data[i++] };
185 float value;
